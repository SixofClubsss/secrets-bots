package main

import "fmt"

func printInitMessage(title string, vars ...interface{}) {
	fmt.Printf("Initializing %s variables:\n", title)
	for i := 0; i < len(vars); i += 2 {
		fmt.Printf("%s: %v\n", vars[i], vars[i+1])
	}
	fmt.Printf("%s variables initialized.\n", title)
}

func printDiscordInit() {
	printInitMessage("Discord", "BotToken", BotToken, "GuildID", GuildID, "AppID", AppID)
}

func printDEROInit() {
	printInitMessage("DERO",
		"DeroServerIP", DeroServerIP, "DeroWalletPort", DeroWalletPort,
		"DeroNodePort", DeroNodePort, "DeroUser", DeroUser,
		"DeroPass", DeroPass, "PongDir", PongDir,
		"PongDB", PongDB, "IAddressTextFile", IAddressTextFile)
}

func printChatGPTInit() {
	printInitMessage("ChatGPT", "ChatGptApi", ChatGptApi)
}
